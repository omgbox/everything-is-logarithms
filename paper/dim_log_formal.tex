\begin{abstract}
We establish a rigorous correspondence between the dimension operator $\dim_K$ in linear algebra and the logarithm. For finite-dimensional vector spaces over finite fields, we prove that $\dim_K V = \log_{|K|} |V|$ exactly, and that all standard dimension axioms follow from logarithmic identities. We also explore the emergence of fractional dimensions when comparing vector spaces over different fields.
\end{abstract}

\section{Motivation}
\label{sec:motivation}
%======================================================================

The dimension of a vector space $V$ over a field $K$ is the cardinality of its basis. Simultaneously, the cardinality of $V$ satisfies $|V| = |K|^{\dim_K V}$ for finite fields. Taking logarithms:
\[
\dim_K V = \log_{|K|} |V|.
\]
We show this is not merely a formula but an \emph{isomorphism} between the algebra of dimensions and the algebra of logarithms.

%======================================================================
\section{Definitions}
\label{sec:definitions}
%======================================================================

\begin{definition}[Vector Space Cardinality]
Let $V$ be a finite-dimensional vector space over a finite field $K = \text{GF}(q)$ with $|K| = q$. Then:
\[
|V| = q^{\dim_K V} = |K|^{\dim_K V}.
\]
\end{definition}

\begin{definition}[Dimension-Logarithm]
The \textbf{dimension} of $V$ over $K$ can be expressed as:
\[
\dim_K V = \log_{|K|} |V| = \frac{\ln |V|}{\ln |K|}.
\]
\end{definition}

\begin{definition}[Baseless Dimension]
Following the baseless logarithm framework, we define:
\[
\dim V = \log |V|
\]
as an abstract object (analogous to $\log x$), with the based dimension:
\[
\dim_K V = \frac{\dim V}{\dim K} = \frac{\log |V|}{\log |K|}.
\]
\end{definition}

\section{Main Theorems}
\label{sec:theorems}
%======================================================================

\begin{theorem}[Dimension-Logarithm Identity]
\label{thm:main}
For any finite-dimensional vector space $V$ over a finite field $K$:
\[
\dim_K V = \log_{|K|} |V|.
\]
\end{theorem}

\begin{proof}
By definition, $|V| = |K|^{\dim_K V}$. Taking $\log_{|K|}$ of both sides:
\[
\log_{|K|} |V| = \log_{|K|} (|K|^{\dim_K V}) = \dim_K V. \qedhere
\]
\end{proof}

\begin{theorem}[Direct Sum = Addition]
\label{thm:directsum}
For vector spaces $U, V$ over $K$:
\[
\dim_K(U \oplus V) = \dim_K U + \dim_K V.
\]
This corresponds to the logarithmic identity:
\[
\log_{|K|}(|U| \cdot |V|) = \log_{|K|} |U| + \log_{|K|} |V|.
\]
\end{theorem}

\begin{proof}
$\dim_K(U \oplus V) = \dim_K U + \dim_K V$ by standard linear algebra. Using Theorem~\ref{thm:main}:
\[
\log_{|K|}(|U| \cdot |V|) = \log_{|K|}(|U|) + \log_{|K|}(|V|) = \dim_K U + \dim_K V. \qedhere
\]
\end{proof}

\begin{theorem}[Tensor Product = Multiplication]
\label{thm:tensor}
For vector spaces $U, V$ over $K$:
\[
\dim_K(U \otimes_K V) = \dim_K U \cdot \dim_K V.
\]
This corresponds to the logarithmic identity:
\[
\log_{|K|}(|U|^{\dim_K V}) = \dim_K V \cdot \log_{|K|} |U|.
\]
\end{theorem}

\begin{proof}
$\dim_K(U \otimes_K V) = \dim_K U \cdot \dim_K V$ by standard linear algebra. The cardinality satisfies:
\[
|U \otimes_K V| = |K|^{\dim_K U \cdot \dim_K V}.
\]
Taking logarithms: $\log_{|K|} |U \otimes_K V| = \dim_K U \cdot \dim_K V$.
\end{proof}

\begin{theorem}[Quotient = Subtraction]
\label{thm:quotient}
For a subspace $W \subseteq V$:
\[
\dim_K(V/W) = \dim_K V - \dim_K W.
\]
This corresponds to:
\[
\log_{|K|}(|V|/|W|) = \log_{|K|} |V| - \log_{|K|} |W|.
\]
\end{theorem}

\begin{proof}
Standard result: $\dim_K(V/W) = \dim_K V - \dim_K W$. The logarithmic form follows from $\log(a/b) = \log a - \log b$.
\end{proof}

\begin{theorem}[Zero Space]
\label{thm:zero}
$\dim_K(\{0\}) = 0$, corresponding to $\log_{|K|}(1) = 0$.
\end{theorem}

\section{Fractional Dimensions}
\label{sec:fractional}
%======================================================================

\begin{definition}[Cross-Field Dimension]
Given fields $K \subseteq L$ (not necessarily $K$-vector spaces), we define:
\[
\dim_{L}(K) = \log_{|L|} |K|.
\]
When $|K| < |L|$, this gives a fractional dimension.
\end{definition}

\begin{proposition}[Half-Dimensions]
\label{prop:half}
For any prime $p$:
\[
\dim_{\text{GF}(p^2)} \text{GF}(p) = \log_{p^2}(p) = \frac{1}{2}.
\]
\end{proposition}

\begin{proof}
\[
\log_{p^2}(p) = \frac{\ln p}{\ln(p^2)} = \frac{\ln p}{2 \ln p} = \frac{1}{2}. \qedhere
\]
\end{proof}

\begin{proposition}[Extension Field Dimension]
\label{prop:extension}
For any prime $p$ and positive integer $n$:
\[
\dim_{\text{GF}(p)} \text{GF}(p^n) = \log_p(p^n) = n.
\]
\end{proposition}

\begin{proof}
$\text{GF}(p^n)$ is an $n$-dimensional vector space over $\text{GF}(p)$, so $\dim_{\text{GF}(p)} \text{GF}(p^n) = n$. The logarithmic form gives $\log_p(p^n) = n$.
\end{proof}

\section{The Dimension Logarithm as an Isomorphism}
\label{sec:isomorphism}
%======================================================================

\begin{theorem}[Isomorphism of Algebraic Structures]
The map $\dim: V \mapsto \log |V|$ is an isomorphism from the semiring of vector spaces (under $\oplus$ and $\otimes$) to the semiring $(\mathbb{R}_{\geq 0}, +, \times)$:
\begin{align*}
\dim(U \oplus V) &= \dim U + \dim V \\
\dim(U \otimes V) &= \dim U \times \dim V \\
\dim(\{0\}) &= 0 \\
\dim(K) &= \log |K|
\end{align*}
\end{theorem}

\begin{proof}
Direct verification using Theorems~\ref{thm:directsum}, \ref{thm:tensor}, and \ref{thm:zero}.
\end{proof}

\begin{remark}
This isomorphism means that the entire structure of finite-dimensional linear algebra over finite fields is \emph{encoded} in the arithmetic of logarithms. The ``mysterious'' properties of dimension (additivity for direct sums, multiplicativity for tensor products) are simply the familiar properties of logarithms.
\end{remark}

%======================================================================
\section{Summary}
\label{sec:summary}
%======================================================================

\begin{enumerate}
\item $\dim_K V = \log_{|K|} |V|$ is an identity, not just a formula.
\item All dimension axioms follow from logarithmic identities:
  \begin{itemize}
  \item $\dim(U \oplus V) = \dim U + \dim V$ from $\log(ab) = \log a + \log b$
  \item $\dim(U \otimes V) = \dim U \cdot \dim V$ from $\log(a^b) = b \log a$
  \item $\dim(V/W) = \dim V - \dim W$ from $\log(a/b) = \log a - \log b$
  \end{itemize}
\item Fractional dimensions emerge naturally: $\dim_{\text{GF}(p^2)} \text{GF}(p) = 1/2$.
\item The map $\dim$ is a semiring isomorphism.
\end{enumerate}
